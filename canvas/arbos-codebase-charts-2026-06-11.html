internal/tool/registry.go — the seam the whole tool catalog plugs into

// Spec declares a tool: its identity, whether it is read-only
// (parallel-safe), a zero-value of its argument struct, and
// the decode+invoke closure.
type Spec struct {
	Name        string
	Description string
	ReadOnly    bool
	ArgSample   any
	Invoke      func(ctx context.Context, raw json.RawMessage) (Result, error)
}
